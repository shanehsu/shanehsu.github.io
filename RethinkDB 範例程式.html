RethinkDB 範例程式

首先是我們的後端應用程式
'use strict'
const express = require('express')
const bodyparser = require('body-parser')
const r = require('rethinkdb')
// Express 應用程式以及中介軟體
const app = express()
// 處理 application/json 的中介軟體
app.use(bodyparser.json())
app.use(function(req, res, next) {
  res.header("Access-Control-Allow-Origin", "*")
  res.header("Access-Control-Allow-Headers", "X-Requested-With")
  next()
})
// 伺服器
const options = {
  host: 'localhost',
  port: 28015
}
// r.connect 回傳一個 Promise，當成功連線時，會將連線傳回。
r.connect(options)
 // 記錄這個連線，等下指令會用到
 .then(establishedConnection => {
   app.locals.rethink_connection = establishedConnection
   
   // 當關閉時，試圖重新連線
   app.locals.rethink_connection.on('close', () => {
     app.locals.rethink_ready = false
     app.locals.rethink_connection.reconnect({
       noreplyWait: true
     }).then(() => {
       app.locals.rethink_ready = true
     }).catch(console.error)
   })
 })
 .then((connection) => {
   // dbList() 回傳的是類似陣列的物件
   r.dbList().run(app.locals.rethink_connection).then(db_list => {
     if (db_list.indexOf('todo') < 0) {
       console.log(`因為沒有 'todo' 資料庫，所以應用程式將建立 'todo' 資料庫。`)
       r.dbCreate('todo').run(app.locals.rethink_connection).then(() => {
         console.log(`資料庫 'todo' 已經建立，準備建立 'default' 資料表。`)
         r.db('todo').createTable('default').run(app.locals.rethink_connection)
          .then(() => {
            console.log(`資料表 'default' 已經建立。`)
            app.locals.rethink_ready = true
          }).catch(console.error)
       }).catch(console.error)
     } else {
       // 資料庫已經存在，檢查資料表
       r.db('todo').tableList().run(app.locals.rethink_connection)
        .then(table_list => {
          if (table_list.indexOf('default') < 0) {
            // 資料表不存在
            r.db('todo').tableCreate('default').run(app.locals.rethink_connection)
             .then(() => {
               console.log(`資料表 'default' 已經建立。`)
               app.locals.rethink_ready = true
            }).catch(console.error)
          } else {
            app.locals.rethink_ready = true
          }
        }).catch(console.error)
     }
   }).catch(console.error)
 })
 .catch(console.error)
app.use((req, res, next) => {
  if (!app.locals.rethink_ready) {
    next(new Error('資料庫尚未就緒。'))
  } else {
    next()
  }
})
app.get('/', (req, res, next) => {
  r.db('todo').table('default').run(app.locals.rethink_connection)
   .then(todos_cursor => {
    todos_cursor.toArray().then(todos => res.json(todos)).catch(console.error)
  })
   .catch(console.error)
})
app.post('/', (req, res, next) => {
  if (req.body.title) {
    r.db('todo').table('default').insert({
      // 強制轉型為字串
      title: '' + req.body.title,
      done: false
    }, {
      returnChanges: true
    }).run(app.locals.rethink_connection).then(changes => {
      console.log('changes')
      if (changes.inserted == 1) {
        res.send(changes.generated_keys[0])
      } else {
        next(new Error('資料庫 API 邏輯錯誤，無法建立待辦事項。'))
      }
    }).catch(console.error)
  } else {
    next(new Error('POST / 必須提供待辦事項標題。'))
  }
})
app.patch('/:id', (req, res, next) => {
  const todo_id = req.params.id
  let update_body = {}
  
  if (req.body.title != undefined) {
    update_body.title = '' + req.body.title
  }
  if (req.body.done != undefined) {
    update_body.done = ((req.body.done) == true)
  }
  
  r.db('todo').table('default').get(todo_id).update(update_body).run(app.locals.rethink_connection)
   .then(() => res.sendStatus(200)).catch(console.error)
})
app.delete('/:id', (req, res, next) => {
  const todo_id = req.params.id
  
  r.db('todo').table('default').get(todo_id).delete().run(app.locals.rethink_connection)
   .then(() => res.sendStatus(200)).catch(console.error)
})
app.listen(3000)
接下來是我們的網頁（很快很髒的用 jQuery 直接來）
<!DOCTYPE html>
<html>
  <head>
    <meta charset="UTF-8">
    <title></title>
    <script src="https://code.jquery.com/jquery-2.2.1.min.js" charset="utf-8"></script>
    <script>
    $(function() {
      load_todos()
    })
    
    function load_todos() {
      $.get('http://localhost:3000/', null, $.noop, 'json').done(function(data) {
        $('#todos').html("")
        const checked = '☑'
        const unchecked = '☐'
        
        for (var index in data) {
          var todo = data[index]
          $('#todos').append('<li>' +
                             '<a onclick="delete_todo(this)" data-id="' + todo.id + '">' +
                             '&times;  ' +
                             '</a>' +
                             '<a onclick="toggle_done(this)" data-id="' + todo.id + '" data-done="' + todo.done + '">' +
                             (todo.done ? checked : unchecked) + '</a>' + ' ' + 
                             '<a onclick="edit_todo(this)" data-id="' + todo.id + '">' +
                             todo.title + 
                             '</a>' +
                             '</li>')
        }
      }).fail(console.error)
    }
    
    function new_todo() {
      var title = $('#todo_title').val()
      var data = JSON.stringify({
        "title": title
      })
      $.ajax('http://localhost:3000/', {
        contentType: 'application/json',
        data: data,
        dataType: 'text',
        method: 'POST'
      }).done(function(data) {
        console.log('成功新增待辦事項。')
        load_todos()
      }).fail(console.error)
    }
    
    function todo_title_changed(element) {
      if ($('#todo_title').val() != '') {
        $('#todo_add_button').prop('disabled', false)
      } else {
        $('#todo_add_button').prop('disabled', true)
      }
    }
    
    function toggle_done(element) {
      var new_todo_done = !(element.getAttribute('data-done') == 'true')
      var id = element.getAttribute('data-id')
      var data = JSON.stringify({
        done: new_todo_done
      })
      $.ajax('http://localhost:3000/' + id, {
        contentType: 'application/json',
        data: data,
        method: 'PATCH'
      }).done(function(data) {
        console.log('成功更新待辦事項。')
        load_todos()
      }).fail(console.error)
    }
    
    function delete_todo(element) {
      var id = element.getAttribute('data-id')
      $.ajax('http://localhost:3000/' + id, {
        method: 'DELETE'
      }).done(function(data) {
        console.log('成功刪除待辦事項。')
        load_todos()
      }).fail(console.error)
    }
    
    function edit_todo(element) {
      var new_title = prompt('新的標題？\n單行，超過一行會被忽略，不可留空。', element.innerHTML).split('\n')[0]
      if (new_title == '') return
      var id = element.getAttribute('data-id')
      var data = JSON.stringify({
        title: new_title
      })
      $.ajax('http://localhost:3000/' + id, {
        contentType: 'application/json',
        data: data,
        method: 'PATCH'
      }).done(function(data) {
        console.log('成功更新待辦事項。')
        load_todos()
      }).fail(console.error)
    }
    </script>
    <style>
    ul {
      list-style-type: none;
    }
    </style>
  </head>
  <body>
    <ul id="todos">
    </ul>
    <input id="todo_title" placeholder="新代辦事項標題" onkeypress="todo_title_changed()">
    <input id="todo_add_button" type="button" onclick="new_todo()" value="新增" disabled="">
  </body>
</html>
