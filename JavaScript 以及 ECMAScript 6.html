JavaScript 以及 ECMAScript 6

JavaScript 基本語法（ECMAScript 5 部分，與大部分瀏覽器相容）

變數、型態

首先，JavaScript 的變數無法定義型態，在第一次給值的時候，會自動產生型態。

// Number 型態
var someNumber = 123
someNumber = 12.23
// String 型態
var someString = "string"
someString = 'someString'
// undefined 值
var someUndefined
// 陣列
var array = [1, 2, 3]
// 字典
var dict = {
  a: 1,
  b: 2
}

用 var 定義出來的變數若不是全域變數則為函數變數，是跟著函數跑的，而不是跟著大括號。

var global_variable = 123
function f() {
  var t = 1
  while (t--) {
    var x = 4
  }
  // 在這裡可以用 x
}
// 不能使用 x 了

運算子

運算方面包含最基本的 +、-、*、/、%、++、--。

指定方面包含最基本的 =、+=、-=、*=、/=、%=。

比較方面包含 ==、===、!=、!==、>、<、>=、<= 以及 ? :。

型態方面包含 typeof 以及 instanceof。

其中，若用 == 比較，則 1 == "1"，若要保證型態相同，則應使用 ===。

typeof 以字串回傳型態，instanceof 檢查物件是否為類別的實例。

函數

JavaScript 的函數非常簡單

function add(a, b) {
  return a + b
}

函數本身其實也是物件

var f = function(a, b) {
  return a + b
}
var argumentCount = f.length()
var x = f(1, 2)

函數的 bind(thisArg, arg1, arg2, ...) 可以用來製造新的函數

this.x = 1
var obj = {
  x: 2,
  t: function (y = 0) {
    return this.x + y
  }
}
obj.t() // 2
var f = obj.t
f() // 1
var f2 = f.bind(obj)
f2() // 2
var f3 = function(a, b) {
  return a + b
}
var f4 = f3.bind(undefined, 4) // f4 = function(b) {return b + 4}
f4(2) // 6

JavaScript 函數所意指的 this 跟在哪裡呼叫那個函數有關係，並不是跟著函數被定義的位置跑。可以用 bind() 指定這個函數的 this 指的是哪個物件。

回傳函數

function factory(a) {
  return function(b) {
    return a + b
  }
}
var t = factory(4)
t(5) // 9

物件

JavaScript 中的物件非常簡單，就是一個字典

var car = {
  manufacturer: 'Volvo',
  model: 'V40',
  name: function() {
    return this.manufacturer + " " + this.model
  }
}
var car_name = car.name()

JavaScript 的類別是以原形為主，物件可以不用為類別的實例，但是也可以從類別來建立物件

var Car = function(manufacturer, model) {
  this.manufacturer = manufacturer
  this.model = model
  this.name = function() {
    return this.manufacturer + " " + this.model
  }
}
var car = new Car('Volvo', 'XC60')

也可以直接修改原形

Car.prototype.name = function() {
  return this.model
}

字串的方法以及屬性

確認內容

  • includes(target)
  • startsWith(target)
  • endsWith(target)

在字串中搜尋

  • indexOf(target, indexBegin = 0)
  • lastIndexOf(target, indexBegin = 0)
  • search(target, indexBegin = 0)
  • search(pattern, indexBegin = 0)

擷取部分內容

  • slice(begin, end)
  • substring(begin, end)
  • substr(begin, length)
  • trim()

slice 的 begin 以及 end 若為負數，則會用倒數的方式。
trim 會刪除開頭結尾的空白字元。

取代內容

  • replace(target, content)
  • replace(pattern, content)

轉換大小寫

  • toUpperCase()
  • toLowerCase()

取得字元

  • charAt(index)
  • charCodeAt(index)

轉換成陣列

  • split(splitPoint)

若用 split('') 則會轉換為字元陣列。

正規表示式（Regular Expression）

var expression = /exp/
expression = /exp/igm

第一個部分是樣式本身，第二個部分是修飾子。

修飾子 意義
i 不論大小寫。
g 搜尋整個字串。
m 進行跨行搜尋。
u 進行 Unicode 搜尋。

用 g 修飾子在字串的 replace 方法中，可以在整個字串中進行取代。

樣式本身可能包含

樣式 意義
[ab] 比對 [] 中的字元。
[a-z] 比對 [] 中的字元區間的其中一個字元。
\d 比對一個數字字元。
\s 比對一個空白字元（包含 tab 等）
\b 比對單字的起始或是結尾。
\uxxxx 比對一個 Unicode 字元。
x+ 比對一個以上的 x。
x* 比對空字串或是一個以上的 x。
x? 比對空字串或是一個 x。

更多資訊：http://www.w3schools.com/jsref/jsref_obj_regexp.asp

錯誤處理

try {
  throw new Error('錯誤訊息')
} catch (error) {
  console.error(error)
} finally {
  console.log('finally 中的內容一定會被執行。')
}

嚴格模式

在 JavaScript 的第一行寫 "use strict" 可以啟用嚴格模式，直譯器將會更嚴格的執行程式碼檢查。也可以只為一個函數啟用嚴格模式。

for...in 迴圈

var object = {
  a: 1,
  b: 2
}
for (var property in object) {
  console.log(property, object[property])
}

[ES6] 箭頭匿名函數

許多 JavaScript 的內建函數必須傳入函數

var array = [1, 2, 3, 4, 5]
function transformFunction(value) {
  return value + 5
}
var newArray = array.map(transformFunction) // [6, 7, 8, 9, 10]

若使用匿名函數，可以讓程式碼更簡潔

var array = [1, 2, 3, 4, 5]
var newArray = array.map(function(value) {
  return value + 5
})

箭頭匿名函數提供一個更簡潔的寫法

var array = [1, 2, 3, 4, 5]
var newArray = array.map(value => {
  return value + 5
})

若匿名函數本身只有一個回傳敘述，可以化簡為

var array = [1, 2, 3, 4, 5]
var newArray = array.map(value => value + 5)

箭頭函數若沒有參數或是有多個參數則必須用括號將參數列表括起來

var array = [1, 2, 3, 4, 5]
var sum = array.reduce((partialSum, value) => partialSum + value)
var one_factory = () => {
  return 1
}

[ES6] 類別

雖然 ECMAScript 6 的類別與 ECMAScript 5 的類別一樣是使用原型，但在 ECMAScirpt 6 中，可以用與 C++ 類似的方法。

class Base {
  constructor(value) {
    this.value = value
  }
}
class Derived extends Base {
  constructor() {
    super(5)
    this.factory = Derived.defaultFactory()
  }
  
  get factoryFunction() {
    return (value) => this.value + this.factory
  }
  
  set specification(specification) {
    this.factory = specification
  }
  
  static defaultFactory() {
    return 0
  }
}

[ES6] 增強的物件常值

var factoryDefault = 0
var object = {
  __proto__: Base.prototype,
  factoryDefault, // 等義於 factoryDefault: factoryDefault
  toString() {
    return 'Base'
  },
  ['a' + 'b']: 41 // ab: 41
}

[ES6] 樣板字串

樣板字串可以用來建立多行的字串。

var str = `Multiline
string is now possible.`
var x = 2
str = `${x}` // "2"

[ES6]解構

var array = [1, 2, 3, 4, 5]
var [one, two, three, four, five] = array
var object = {
  a: 1,
  b: 2
}
var {
  a: _one,
  b: _two
} = object
// 用來實作函數的有名參數
function f({
  parameter1: a,
  parameter2: b
}) {
  return a + b
}
f({
  parameter2: 1,
  parameter1: 2
})

安全性，若解構的資料大小有可能不一樣，你應該使用

var [a] = [] // a === undefined
var [a = 1] = [] // a === 1

[ES6] 函數參數

函數參數可以有預設值（Default）、動態長度（Rest）、或是在呼叫函數的時候將陣列的內容分散（Spread）出去。

// Default
function f_1(x, y = 12) {
  return x + y
}
f_1(1, undefined)
f_1(1)
// ES5 Arguments
function f_2() {
  return arguments.length
}
f_2(1, 2, 3, 4, 5) // 5
// Rest
function f_3(x, ...y) {
  return x * y.length
}
f_3(1, 2, 3, 4) // 1 * 3
// Spread
var array = [2, 3, 4, 5, 6]
f_3(...array) // 2 * 4

[ES6] let 以及 const

原本 ECMAScript 5 的 var 定義出的是全域變數或是函數範圍的變數（在整個函數內可用）。

ECMAScript 6 加入 let，將該變數的範圍定義在大括號中；以及 const 定義出只能被給值一次。

"use strict"
function f() {
  let x
  {
    const x = 'hello'
    x = 'foo' // 錯誤，被給值第二次了！
  }
  let x = 'inner' // 錯誤，被定義第二次了！
}

[ES6] 迭代器以及 for...of 迴圈

let fibonacci = {
  [Symbol.iterator]() {
    let pre = 0, cur = 1
    return {
      next() {
        [pre, cur] = [cur, pre + cur]
        return {
          done: false,
          value: cur
        }
      }
    }
  }
}
for (var n of fibonacci) {
  console.log(n)
}

當然，for...of 迴圈也可以用在陣列上

let array = [1, 2, 3, 4, 5]
for (var element of array) {
  console.log(element)
}
for (var index in array) {
  console.log(array[index])
}

[ES6] 產生器

var fibonacci = {
  [Symbol.iterator]: function*() {
    let pre = 0, cur = 1
    while (true) {
      [pre, cur] = [cur, pre + cur]
      yield cur
    }
  }
}

[ES6] Unicode 字元

let x = "吉祥"
let length = x.length // 4
let codePoint = x.codePointAt(0)
for (let c of x) {
  console.log(c)
}

[ES6] 資料結構

ECMAScript 6 提供 Map、Set、WeakMap 以及 WeakSet。

其中 Weak 代表當它所包含的資料只由它所擁有，則會被自動釋放。換句話說，它不能是該值的唯一擁有者。

var set = new Set()
set.add(1).add(2)
set.has(1)
var map = new Map()
map.set('a', 1)
map.get('a')
var weakmap = new WeakMap()
var k = '2'
weakmap.set('a', 1)
weakmap.set('b', k)
weakmap.size === 1 // 因為 'a': 1 被釋放

[ES6] 保證（Promise）

保證是一個進行非同步處理的方法。原本的方法

http.get(url, function(data, err) {
  if (err) {
    // 錯誤
    return
  }
  
  // 處理資料
})
http.get(url).then(function(data) {
  
}).then(function(data) {
  // 可以有多個處理常式
}).catch(function(err) {
  // 利用 then 或是 catch 區分成功或是失敗
}).catch(function(err) {
  
})

建立 Promise

假設我們要寫一個複雜的亂數程式，需要一段時間才能算出亂數。而且，若算出來的值為 0，則算是失敗。

const random = function() {
  return new Promise(function(resolve, reject) {
    let value = 5
    if (value == 0) {
      reject( new Error('計算失敗'))
    } else {
      resolve(value)
    }
  })
}
random().then(console.log).catch(console.error)
