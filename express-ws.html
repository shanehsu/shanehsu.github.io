express-ws

用 npm install express-ws 安裝套件

express-ws 會修改 app 使得他增加一個 app.ws 方法。

'use strict'
const express = require('express')
const app = express()
const express_ws = require('express-ws')(app)
// 先對 WebSocket 連線做認證
app.get(/.websocket.*$/, function(req, res, next) {
  console.log('收到 WebSocket Upgrade Request 了吧。')
  const random = Math.abs(Math.random()) % 3 + 1
  if (random >= 2) {
    console.log('拒絕了 WebSocket 連線。')
    res.sendStatus(401)
  } else {
    console.log('接受了 WebSocket 連線。')
    next()
  }
})
// 收到的訊息直接 Log 出來 
app.ws('/', (ws, req) => { ws.on('message', console.log) })
 
app.listen(3000)

就這樣，非常簡單。你可以用瀏覽器的 JavaScript 測試他。

var ws = new WebSocket('ws://localhost:3000/')
ws.send('123')

我們的 Node 應用程式應該會記錄出 123。
